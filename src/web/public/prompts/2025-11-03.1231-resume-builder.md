## Technical Implementation Details

### Context-Based Architecture
```typescript
// ResumeContext provides centralized state management
export type ResumeContextType = {
  config: ResumeConfig;
  isDownloading: boolean;
  setConfig: (config: ResumeConfig) => void;
  setIsDownloading: (isDownloading: boolean) => void;
  updateConfig: (updates: Partial<ResumeConfig>) => void;
  updateData: (data: typeof resumeData) => void;
};
```

### Modular Skills System
```typescript
export type SkillSet = {
  title: string;
  description: string;
};

export type SkillCategory = {
  title: string;
  items: SkillSet[];
};

// In ResumeData
skills: SkillCategory[];
```

### Page Indicator Positioning
```typescript
const StyledPageIndicator = styled.div<{ $page: number; $maxPages: number; $show: boolean }>`
  position: absolute;
  top: ${(props) => `${(props.$page / props.$maxPages) * 100}%`};
  visibility: ${(props) => (props.$show ? 'visible' : 'hidden')};

  @media print {
    display: none !important;
    visibility: hidden !important;
  }
`;
```
